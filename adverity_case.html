# Define the template for data aggregation by hour and calculate the CTR
# CTR = number of clicks / number of impressions
# - number of clicks is calculated by summing up the click-variable
# - number of impression is calculated by *counting* the click-variable

dask_df = data.groupby(['hour'])['click'].agg(['sum', 'count'])
dask_df['CTR'] = dask_df['sum'] / dask_df['count']

# Build the outlier detector as specified in the case description
import numpy as np

def outlier_detector(row):
    x = row['CTR']
    mu = row['CTR_mean']
    sigma = row['CTR_std']
    
    if np.abs(x - mu) / sigma > 1.5:
        return 'Outlier'
    else:
        return 'Inlier'
    
df['datapoint_type'] = [outlier_detector(row) for _, row in df.iterrows()]
